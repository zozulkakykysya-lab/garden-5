<script>
  // We use window.onload to ensure Leaflet library finishes loading in the browser first
  window.onload = function() {
    
    // SETUP FOR FANTASY/CUSTOM IMAGES
    // L.CRS.Simple treats the map as a flat XY pixel grid instead of a geographical globe
    var map = L.map('map', {
      crs: L.CRS.Simple,
      minZoom: -1.6,
      maxZoom: 1
    });

    // CONFIGURATION: Enter your image dimensions and URL
    // Replace these with the actual pixel width and height of your map image
    var w = 2048; 
    var h = 1536;
    
    // Provide the path to your map image file
    // If stored in your vault content folder, use the relative web path (e.g., "/static/map.png")
    var imageUrl = 'https://github.com/airiyourdm/sinnerdnd/blob/main/quartz/static/mapclear.png?raw=true'; 

    // Calculate map boundaries based on image proportions
    var southWest = map.unproject([0, h], map.getMaxZoom()-1);
    var northEast = map.unproject([w, 0], map.getMaxZoom()-1);
    var bounds = new L.LatLngBounds(southWest, northEast);

    // Add the image overlay to the map grid
    L.imageOverlay(imageUrl, bounds).addTo(map);

    // Tell the map to fit the entire image on the screen and center it
    map.setMaxBounds(bounds);
    map.fitBounds(bounds);

    // ==========================================
    // ADDING MARKERS (POINTS OF INTEREST)
    // ==========================================
    
    // Marker 1: Capital City
    // In L.CRS.Simple, coordinates are roughly [Y, X] values from the bottom-left corner
    var markerStart = L.marker([-1400, 400]).addTo(map);
    markerStart.bindPopup("<b>Начало приключения</b><br>Пляж, на котором оказались Грешники.");

    // Marker 2: Dangerous Dungeon
    var markerDungeon = L.marker([-750, 1040]).addTo(map);
    markerDungeon.bindPopup("<b>Сердце Бездны</b><br>Место, где давным-давно герои замечатали Первородный грех.");

    // OPTIONAL HELPER: Click anywhere on the map to see coordinates in the browser console
    // Use this while designing to easily find where to place your markers!
    map.on('click', function(e) {
        console.log("Clicked coordinates: [" + e.latlng.lat + ", " + e.latlng.lng + "]");
    });
  };
</script>
